#include <string>
#include <vector>

using namespace std;

string solution(int n) {
    string answer = "";
    
    for(int i = 0;i<n;i++){
        if(i%2 == 0){// 짝수인 경우
            answer.append("수");
        }
        else{// 홀수인 경우
            answer.append("박");
        }
    }
    return answer;
}
