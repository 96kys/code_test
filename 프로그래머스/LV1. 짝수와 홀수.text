#include <string>
#include <vector>

using namespace std;

string solution(int num) {
    string answer = "";
    

    
    if(num == 0){// 0에 관해서
        answer = "Even";
    }
    else if(num%2 == 0){//홀수
        answer = "Even";
    }
    else{//짝수
        answer = "Odd";
    }
    
    return answer;
}
